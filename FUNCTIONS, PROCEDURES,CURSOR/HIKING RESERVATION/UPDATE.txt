CREATE OR REPLACE PROCEDURE HIKE_RESERVE_LIST_UPDATE (
   V_NAME              VARCHAR,
   V_NUM_OF_JOINERS    NUMBER,
   V_MOUNTAIN_PICK     VARCHAR,
   V_DATE_OF_HIKE      VARCHAR)
IS
BEGIN
   UPDATE HIKE_RESERVATION
      SET NUM_OF_JOINERS = V_NUM_OF_JOINERS,
          MOUNTAIN_PICK = V_MOUNTAIN_PICK,
          DATE_OF_HIKE = V_DATE_OF_HIKE
    WHERE NAME = V_NAME;

   COMMIT;
   DBMS_OUTPUT.PUT_LINE ('UPDATED.');
END;
