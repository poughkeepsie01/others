SELECT JOB_ID, COUNT(EMPLOYEE_ID) NUMOFEMP FROM EMPLOYEES
WHERE FIRST_NAME LIKE 'S%'
AND HIRE_DATE > TO_DATE('012021','MMYYYY')
GROUP BY JOB_ID 


SELECT MAX(EMPLOYEE_ID) EMP, JOB_ID, HIRE_DATE  FROM EMPLOYEES
WHERE HIRE_DATE BETWEEN TO_DATE ('01012020','MMDDYYYY') AND TO_DATE('12012020','MMDDYYYY')
GROUP BY JOB_ID,
         HIRE_DATE
ORDER BY EMP DESC
